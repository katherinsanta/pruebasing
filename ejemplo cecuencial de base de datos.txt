
  SELECT 
      a.direccionOrig
      ,a.direccionDest 
      ,a.nombrePasajero
      ,convert(char (17),a.fechaServicio,110)as 'fechaservicio'         
      ,convert (char(5),a.horaServicio,108)as 'hora servicio' 
      ,b.nombreEstado
      ,a.valeAsignado            
      ,a.movil
      ,a.placa
      ,a.nombreconductor
   FROM PDV_SolicitudesTransporte a  inner join dbo.PDV_TiposEstados b 
   on a.estado=b.idestado
   left join conductoresall c on a.idconductor=c.idconductor
   where    
   CONVERT(DATETIME,CONVERT(CHAR(10),a.fechaServicio,120) + ' ' + CONVERT(CHAR(5),horaServicio,108)) >= DATEADD(mi,-45,GETDATE()) 
   and
   CONVERT(DATETIME,CONVERT(CHAR(10),a.fechaServicio,120) + ' ' + CONVERT(CHAR(5),horaServicio,108)) <= DATEADD(mi,75,GETDATE())
   and a.estado in (2,9)
order by a.horaServicio
